3-topshiriq

Quyidagi text ning faqat lotin tilidagi harflarining Unicode ni toping. 

String text = "2020  PDP Onlineㅋㅋㅋ";




void main() {
  String text = "2020  PDP Onlineㅋㅋㅋ";
  var input = '';
  List list = [];
  var item1;
  list = text.codeUnits;

  for (var item in list) {
    if (item >= 65 && item <= 122 || item == 32) {
      item1 = new String.fromCharCode(item);
      input = input + item1;
    }
  }
  print(input);
}
